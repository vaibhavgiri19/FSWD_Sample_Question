MovieCard.jsx (Solution)

import React from "react";

function MovieCard({ title, director, year, genre, isAvailable }) {
  return (
    <div className="movie-card">
      <h3>{title}</h3>
      <p>Director: {director}</p>
      <p>Year: {year}</p>
      <p>Genre: {genre}</p>

      {isAvailable ? (
        <p className="status-available">Available</p>
      ) : (
        <p className="status-unavailable">Unavailable</p>
      )}
    </div>
  );
}

export default MovieCard;
