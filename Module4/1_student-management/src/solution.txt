StudentCard.jsx Solution:

import React from "react";

function StudentCard({ name, grade, subject, email, isHonorRoll }) {
  return (
    <div className="student-card">
      <h3>{name}</h3>
      <p>Subject: {subject}</p>
      <p>Grade: {grade}</p>
      <p>{email}</p>
      {isHonorRoll && (
        <span className="honor-badge">Honor Roll</span>
      )}
    </div>
  );
}

export default StudentCard;



App.jsx Solution:

import React from "react";
import StudentCard from "./components/StudentCard";

function App() {
  const students = [
    {
      id: 1,
      name: "Alice Johnson",
      grade: "A",
      subject: "Mathematics",
      email: "alice@school.edu",
      isHonorRoll: true
    },
    {
      id: 2,
      name: "Bob Lee",
      grade: "B",
      subject: "History",
      email: "bob@school.edu",
      isHonorRoll: false
    },
    {
      id: 3,
      name: "Cathy Brown",
      grade: "A+",
      subject: "Physics",
      email: "cathy@school.edu",
      isHonorRoll: true
    },
    {
      id: 4,
      name: "Dan Smith",
      grade: "C",
      subject: "Chemistry",
      email: "dan@school.edu",
      isHonorRoll: false
    }
  ];

  return (
    <div>
      <h1>Student Management</h1>
      <div className="student-grid">
        {students.map((student) => (
          <StudentCard
            key={student.id}
            name={student.name}
            grade={student.grade}
            subject={student.subject}
            email={student.email}
            isHonorRoll={student.isHonorRoll}
          />
        ))}
      </div>
    </div>
  );
}

export default App;
