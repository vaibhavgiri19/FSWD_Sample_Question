import React from 'react';

function BookCard({ title, author, year, genre, isAvailable }) {
  return (
    <div className="book-card">
      {/* Display book title */}
      <h3>{title}</h3>

      {/* Display author */}
      <p>by {author}</p>

      {/* Display publication year */}
      <p>({year})</p>

      {/* Display genre */}
      <p>Genre: {genre}</p>

      {/* Conditional rendering for availability */}
      {isAvailable ? (
        <p className="status-available">Available</p>
      ) : (
        <p className="status-unavailable">Checked Out</p>
      )}
    </div>
  );
}

export default BookCard;
